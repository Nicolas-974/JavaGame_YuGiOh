package Graveyard;

import java.util.ArrayList;
import java.util.List;
import Card.Card;

/**
 * Classe représentant le Cimetière d’un joueur.
 */
public class Graveyard {

    // --- Attributs ---
    private List<Card> cards;

    // --- Constructeur ---
    public Graveyard() {
        this.cards = new ArrayList<>();
    }

    // --- Méthodes principales ---

    /** Ajoute une carte au Cimetière */
    public void addCard(Card c) {
        cards.add(c);
        System.out.println("Carte envoyée au Cimetière: " + c.getName());
    }

    /** Retire une carte spécifique du Cimetière */
    public boolean removeCard(Card c) {
        return cards.remove(c);
    }

    /** Retourne la dernière carte envoyée au Cimetière */
    public Card getLastCard() {
        if (isEmpty()) return null;
        return cards.get(cards.size() - 1);
    }

    /** Recherche une carte par son nom */
    public Card searchByName(String name) {
        for (Card c : cards) {
            if (c.getName().equalsIgnoreCase(name)) {
                return c;
            }
        }
        return null;
    }

    /** Retourne le nombre de cartes dans le Cimetière */
    public int size() {
        return cards.size();
    }

    /** Vérifie si le Cimetière est vide */
    public boolean isEmpty() {
        return cards.isEmpty();
    }

    /** Retourne toutes les cartes du Cimetière */
    public List<Card> getAllCards() {
        return cards;
    }
}
