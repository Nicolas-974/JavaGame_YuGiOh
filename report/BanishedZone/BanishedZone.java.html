package BanishedZone;

import java.util.ArrayList;
import java.util.List;
import Card.Card;

/**
 * Classe représentant la Zone des cartes bannies (hors jeu).
 */
public class BanishedZone {

    // --- Attributs ---
    private List<Card> cards;

    // --- Constructeur ---
    public BanishedZone() {
        this.cards = new ArrayList<>();
    }

    // --- Méthodes principales ---

    /** Ajoute une carte à la Zone Bannies */
    public void addCard(Card c) {
        cards.add(c);
        System.out.println("Carte bannie: " + c.getName());
    }

    /** Retire une carte spécifique de la Zone Bannies */
    public boolean removeCard(Card c) {
        return cards.remove(c);
    }

    /** Vérifie si une carte donnée est présente dans la Zone Bannies */
    public boolean contains(Card c) {
        return cards.contains(c);
    }

    /** Retourne le nombre de cartes bannies */
    public int size() {
        return cards.size();
    }

    /** Vérifie si la Zone Bannies est vide */
    public boolean isEmpty() {
        return cards.isEmpty();
    }

    /** Retourne toutes les cartes bannies */
    public List<Card> getAllCards() {
        return cards;
    }
}
