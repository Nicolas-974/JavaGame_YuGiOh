package Game;

import Deck.Deck;
import Player.Player;

/**
 * Classe représentant une partie Yu-Gi-Oh!
 * Elle orchestre le déroulement du duel : joueurs, phases, tours, conditions de victoire.
 */
public class Game {

    // --- Attributs ---
    private Player player1;
    private Player player2;
    private Player currentPlayer;
    private Phase currentPhase;
    private int turnCount;
    private boolean isGameOver;

    // --- Constructeur ---
    public Game(Player player1, Player player2) {
        this.player1 = player1;
        this.player2 = player2;
        this.currentPlayer = player1; // par défaut, player1 commence
        this.currentPhase = Phase.DRAW; // première phase du tour
        this.turnCount = 1;
        this.isGameOver = false;
    }

    // --- Méthodes principales ---
    public void startGame() {
        System.out.println("La partie commence entre " + player1.getName() + " et " + player2.getName());
        player1.setDeck(new Deck(/* paramètres */));
        player2.setDeck(new Deck(/* paramètres */));
        currentPlayer = player1;
        player1.drawCard();
        player2.drawCard();
        turnCount = 1;
        isGameOver = false;

    }

    public Phase getCurrentPhase() {
        return currentPhase;
    }

    public void nextPhase() {
        currentPhase = currentPhase.next(); // nécessite enum Phase avec méthode next()
        System.out.println("Phase suivante: " + currentPhase);
    }

    public void endTurn() {
        currentPlayer.endTurn();
        resetPhase();
        switchPlayer();
        incrementTurn();
        System.out.println("Fin du tour. C'est maintenant au tour de " + currentPlayer.getName());
    }

    public Player getOpponent() {
        return (currentPlayer == player1) ? player2 : player1;
    }

    public Player getCurrentPlayer() {
        return currentPlayer;
    }

    public void checkWinCondition() {
        if (!player1.isAlive()) {
            endGame(player2);
        } else if (!player2.isAlive()) {
            endGame(player1);
        }
    }

    public void switchPlayer() {
        currentPlayer = (currentPlayer == player1) ? player2 : player1;
    }

    public int getTurnCount() {
        return turnCount;
    }

    public void incrementTurn() {
        turnCount++;
    }

    public void resetPhase() {
        currentPhase = Phase.DRAW;
    }

    public void endGame(Player winner) {
        isGameOver = true;
        System.out.println("La partie est terminée ! Le vainqueur est " + winner.getName());
    }

    public boolean isOver() {
        return isGameOver;
    }
}
