package Card;
import Board.Board;

/**
 * Classe représentant un effet générique associé à une carte.
 * Décrit le texte et le type de l'effet.
 */
public class Effect {

    /** Texte décrivant l’effet. */
    private String description;

    /** Type d’effet (ex : Continu, Déclencheur, Rapide, etc.). */
    private String effectType;

    /**
     * Constructeur d'un effet.
     *
     * @param description texte décrivant l'effet
     * @param effectType  type d'effet (Continu, Déclencheur, Rapide, etc.)
     */
    public Effect(String description, String effectType) {
        this.description = description;
        this.effectType = effectType;
    }

    /** @return la description de l’effet */
    public String getDescription() {
        return description;
    }

    /** @return le type de l’effet */
    public String getEffectType() {
        return effectType;
    }
