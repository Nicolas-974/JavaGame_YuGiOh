package Card;
import Player.Player;
/**
 * Classe abstraite représentant une carte générique dans le jeu Yu-Gi-Oh!.
 * <p>
 * La classe {@code Card} est la super-classe de toutes les cartes du jeu
 * (Monstres, Magies, Pièges). Elle définit les attributs et comportements
 * communs à toutes les cartes et sert de base pour les spécialisations
 * {@link MonsterCard}, {@link SpellCard}, {@link TrapCard}.
 *
 * <h2>Responsabilités</h2>
 * <ul>
 *   <li>Stocker les informations de base d'une carte (nom, description, identifiant).</li>
 *   <li>Gérer le propriétaire de la carte ({@link Player}).</li>
 *   <li>Gérer l'état recto/verso de la carte.</li>
 *   <li>Fournir une méthode abstraite {@link #activateEffect()} que chaque sous-classe doit implémenter.</li>
 * </ul>
 *
 * <h2>Relations</h2>
 * <ul>
 *   <li>Héritage : {@link MonsterCard}, {@link SpellCard}, {@link TrapCard} héritent de cette classe.</li>
 *   <li>Association : chaque carte appartient à un {@link Player}.</li>
 *   <li>Interaction : les cartes circulent entre les zones de jeu (Deck, Main, Board, Cimetière, Zone Bannie).</li>
 * </ul>
 *
 * @author Nicolas
 * @version 1.0
 */
public abstract class Card {

    /** Nom de la carte. */
    private final String name;

    /** Texte descriptif ou effet de la carte. */
    private final String description;

    /** Identifiant unique de la carte. */
    private final int cardId;

    /** Joueur propriétaire de la carte. */
    private Player owner;

    /** Indique si la carte est posée face verso (true = face cachée). */
    private boolean isFaceDown;

    /**
     * Constructeur de base pour une carte.
     *
     * @param name        le nom de la carte
     * @param description le texte descriptif ou effet de la carte
     * @param cardId      identifiant unique de la carte
     * @param owner       joueur propriétaire de la carte
     */
    public Card(String name, String description, int cardId, Player owner) {
        this.name = name;
        this.description = description;
        this.cardId = cardId;
        this.owner = owner;
        this.isFaceDown = false; // par défaut face recto
    }

    /** @return le nom de la carte */
    public String getName() { return name; }

    /** @return la description de la carte */
    public String getDescription() { return description; }

    /** @return l'identifiant unique de la carte */
    public int getCardId() { return cardId; }

    /** @return le joueur propriétaire de la carte */
    public Player getOwner() { return owner; }

    /**
     * Définit le propriétaire de la carte.
     *
     * @param owner le nouveau propriétaire
     */
    public void setOwner(Player owner) { this.owner = owner; }

    /**
     * Définit si la carte est posée face verso.
     *
     * @param value true si la carte est face verso, false si elle est face recto
     */
    public void setFaceDown(boolean value) { this.isFaceDown = value; }

    /** @return true si la carte est posée face verso, false sinon */
    public boolean isFaceDown() { return isFaceDown; }

    /**
     * Inverse l'état recto/verso de la carte (utile pour les monstres FLIP).
     */
    public void flip() { this.isFaceDown = !this.isFaceDown; }

    /**
     * Méthode abstraite que chaque sous-classe doit implémenter
     * pour définir l'effet spécifique de la carte.
     */
    public abstract void activateEffect();

    /**
     * Retourne une représentation textuelle de la carte.
     * <p>
     * Si la carte est face verso, retourne "Carte face cachée".
     * Sinon, retourne le nom et la description de la carte.
     *
     * @return une chaîne décrivant la carte
     */
    @Override
    public String toString() {
        if (isFaceDown) return "Carte face cachée";
        return name + " - " + description + " (Face-Up)";
    }
}
