package Card;
import Player.Player;


/**
 * Représente une carte Piège dans le jeu Yu-Gi-Oh!.
 * <p>
 * La classe {@code TrapCard} hérite de {@link Card} et ajoute des attributs
 * spécifiques aux cartes Piège, comme leur type (Normal, Contre, Continu),
 * leur effet et leur caractère persistant ou non.
 *
 * <h2>Responsabilités</h2>
 * <ul>
 *   <li>Stocker le type de la carte Piège.</li>
 *   <li>Gérer l’effet associé à la carte.</li>
 *   <li>Indiquer si la carte reste sur le terrain après activation (Piège Continu).</li>
 *   <li>Permettre l’activation de l’effet de la carte.</li>
 * </ul>
 *
 * <h2>Relations</h2>
 * <ul>
 *   <li>Héritage : hérite de la classe abstraite {@link Card}.</li>
 *   <li>Association : chaque carte Piège peut avoir un {@link Effect} spécifique.</li>
 *   <li>Interaction : posée face verso sur le {@code Board}, activée en réponse à des actions.</li>
 *   <li>Interaction : utilisée par un {@link Player} pour contrer, bloquer ou perturber l’adversaire.</li>
 * </ul>
 *
 * @author Nicolas
 * @version 1.0
 */
public class TrapCard extends Card {

    /** Type de la carte Piège (ex : Normal, Contre, Continu). */
    private String trapType;

    /** Effet associé à la carte Piège. */
    private Effect effect;

    /** Indique si la carte Piège reste sur le terrain après activation (ex : Piège Continu). */
    private boolean isContinuous;

    /**
     * Constructeur d’une carte Piège.
     *
     * @param name        nom de la carte
     * @param description description ou effet de la carte
     * @param cardId      identifiant unique
     * @param owner       joueur propriétaire
     * @param trapType    type de la carte Piège (Normal, Contre, Continu)
     * @param effect      effet associé à la carte
     * @param isContinuous true si la carte reste sur le terrain après activation
     */
    public TrapCard(String name, String description, int cardId, Player owner,
                    String trapType, Effect effect, boolean isContinuous) {
        super(name, description, cardId, owner);
        this.trapType = trapType;
        this.effect = effect;
        this.isContinuous = isContinuous;
    }

    /** @return le type de la carte Piège */
    public String getTrapType() { return trapType; }

    /** @return true si la carte Piège est continue (reste sur le terrain après activation) */
    public boolean isContinuous() { return isContinuous; }

    /** @return l’effet associé à la carte Piège */
    public Effect getEffect() { return effect; }

    /**
     * Associe un effet à la carte Piège.
     *
     * @param effect l'effet à lier à cette carte
     */
    public void setEffect(Effect effect) {
        this.effect = effect;
    }

    /**
     * Active l’effet de la carte Piège.
     * <p>
     * Par défaut, affiche une description de l’effet. Désormais,
     * appelle {@code effect.activate(...)} pour exécuter l’effet.
     */
    @Override
    public void activateEffect() {
        if (effect != null) {
            System.out.println("Activation de l'effet de Piège: " + effect.getDescription());
            // Exemple d'appel : la carte elle-même est la source, cible optionnelle
            effect.activate(this, null, getOwner().getBoard());
        } else {
            System.out.println(getName() + " n’a pas d’effet à activer.");
        }
    }

    public void setContinuous(boolean isContinuous) {
        this.isContinuous = isContinuous;
    }

    /**
     * Retourne une représentation textuelle de la carte Piège,
     * incluant son type, son caractère continu et son effet.
     *
     * @return une chaîne décrivant la carte Piège
     */
    @Override
    public String toString() {
        String base = "Duelliste " + getOwner().getName() + " joue: \n" +
                      getName() + " - " + getDescription() +
                      "\nType de Piège: " + trapType;

        if (isContinuous) {
            base += " (Continue)";
        }

        base += " / Password: " + getCardId();
        base += "\nEffet: " + effect.getDescription();

        return base;
    }
}
