package Card;
import Player.Player;


/**
 * Représente une carte Magie dans le jeu Yu-Gi-Oh!.
 * <p>
 * La classe {@code SpellCard} hérite de {@link Card} et ajoute des attributs
 * spécifiques aux cartes Magie, comme leur type (Normal, Jeu-Rapide, Équipement,
 * Terrain, Rituel, etc.), leur effet et leur caractère continu ou non.
 *
 * <h2>Responsabilités</h2>
 * <ul>
 *   <li>Stocker le type de la carte Magie.</li>
 *   <li>Gérer l’effet associé à la carte.</li>
 *   <li>Indiquer si la carte reste sur le terrain après activation (Magie Continue, Terrain, Équipement).</li>
 *   <li>Permettre l’activation de l’effet de la carte.</li>
 * </ul>
 *
 * <h2>Relations</h2>
 * <ul>
 *   <li>Héritage : hérite de la classe abstraite {@link Card}.</li>
 *   <li>Association : chaque carte Magie peut avoir un {@link Effect} spécifique.</li>
 *   <li>Interaction : posée ou activée depuis la main sur le {@code Board} du joueur.</li>
 *   <li>Interaction : utilisée par un {@link Player} pour influencer le terrain, les monstres ou les points de vie.</li>
 * </ul>
 *
 * @author Nicolas
 * @version 1.0
 */
public class SpellCard extends Card {

    /** Type de la carte Magie (ex : Normal, Jeu-Rapide, Équipement, Terrain, Rituel). */
    private String spellType;

    /** Effet associé à la carte Magie. */
    private Effect effect;

    /** Indique si la carte Magie reste sur le terrain après activation (ex : Magie Continue, Terrain, Équipement). */
    private boolean isContinuous;

    /**
     * Constructeur d’une carte Magie.
     *
     * @param name        nom de la carte
     * @param description description ou effet de la carte
     * @param cardId      identifiant unique
     * @param owner       joueur propriétaire
     * @param spellType   type de la carte Magie (Normal, Jeu-Rapide, etc.)
     * @param effect      effet associé à la carte
     * @param isContinuous true si la carte reste sur le terrain après activation
     */
    public SpellCard(String name, String description, int cardId, Player owner,
                     String spellType, Effect effect, boolean isContinuous) {
        super(name, description, cardId, owner);
        this.spellType = spellType;
        this.effect = effect;
        this.isContinuous = isContinuous;
    }

    /** @return le type de la carte Magie */
    public String getSpellType() { return spellType; }

    /** @return true si la carte est continue (reste sur le terrain après activation) */
    public boolean isContinuous() { return isContinuous; }

    /** @return l’effet associé à la carte Magie */
    public Effect getEffect() { return effect; }

    /** 
     * Associe un effet à la carte Magie.
     * 
     * @param effect l'effet à lier à cette carte
     */
    public void setEffect(Effect effect) {
        this.effect = effect;
    }

    /**
     * Active l’effet de la carte Magie.
     * <p>
     * Par défaut, affiche une description de l’effet. Désormais,
     * appelle {@code effect.activate(...)} pour exécuter l’effet.
     */
    @Override
    public void activateEffect() {
        if (effect != null) {
            System.out.println("Activation de l'effet de Magie: " + effect.getDescription());
            // Exemple d'appel : la carte elle-même est la source, pas forcément de cible
            effect.activate(this, null, getOwner().getBoard()); 
        } else {
            System.out.println(getName() + " n’a pas d’effet à activer.");
        }
    }

    /**
     * Retourne une représentation textuelle de la carte Magie,
     * incluant son type, son caractère continu et son effet.
     *
     * @return une chaîne décrivant la carte Magie
     */
    @Override
    public String toString() {
        String base = "Duelliste " + getOwner().getName() + " joue: \n" +
                      getName() + " - " + getDescription() +
                      "\nType de Magie: " + spellType;

        if (isContinuous) {
            base += " (Continue)";
        }

        base += " / Password: " + getCardId();
        base += "\nEffet: " + effect.getDescription();

        return base;
    }
}
