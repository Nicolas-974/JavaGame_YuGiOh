package Hand;

import java.util.ArrayList;
import java.util.List;
import Card.Card;

/**
 * Classe représentant la main d’un joueur.
 * Elle gère les cartes actuellement disponibles pour être jouées, posées ou défaussées.
 */
public class Hand {

    // --- Attributs ---
    private List<Card> cards;

    // --- Constructeur ---
    public Hand() {
        this.cards = new ArrayList<>();
    }

    // --- Méthodes principales ---

    /** Ajoute une carte à la main (souvent après une pioche) */
    public void addCard(Card c) {
        cards.add(c);
    }

    /** Retire une carte spécifique de la main */
    public boolean removeCard(Card c) {
        return cards.remove(c);
    }

    /** Récupère une carte par son index */
    public Card getCard(int index) {
        if (index < 0 || index >= cards.size()) {
            return null;
        }
        return cards.get(index);
    }

    /** Retourne toutes les cartes de la main */
    public List<Card> getAllCards() {
        return cards;
    }

    /** Retourne le nombre de cartes en main */
    public int size() {
        return cards.size();
    }

    /** Vérifie si la main est vide */
    public boolean isEmpty() {
        return cards.isEmpty();
    }

    /** Vérifie si une carte donnée est présente en main */
    public boolean contains(Card c) {
        return cards.contains(c);
    }

    /** Défausse une carte (en l’envoyant au Cimetière) */
    public boolean discard(Card c) {
        // Pour l’instant, on simule juste la défausse en retirant la carte
        return removeCard(c);
    }

    /** Vide complètement la main (utile en cas de reset de partie) */
    public void clear() {
        cards.clear();
    }
}
