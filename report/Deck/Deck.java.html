package Deck;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

import Card.Card;
import Card.MonsterCard;
import Card.SpellCard;
import Card.TrapCard;
import Card.Effect;
import Player.Player;

/**
 * Classe représentant la pioche principale d’un joueur.
 */
public class Deck {

    // --- Attributs ---
    private List<Card> cards;

    // --- Constructeurs ---
    public Deck(List<Card> initialCards) {
        this.cards = new ArrayList<>(initialCards);
    }

    public Deck() {
        // initialisation minimale, par exemple une liste vide
        this.cards = new ArrayList<>();
    }

    /**
     * Constructeur qui crée un deck par défaut avec quelques cartes de test.
     * @param owner joueur propriétaire des cartes
     */
    public Deck(Player owner) {
        this.cards = new ArrayList<>();

        // Exemple de monstre normal
        cards.add(new MonsterCard(
            "Blue-Eyes White Dragon",
            "Legendary dragon",
            1,
            owner,
            3000, 2500, 8,
            "LIGHT", "Dragon", "Normal"
        ));

        // Exemple de monstre avec effet
        Effect reviveEffect = new Effect("Revive a monster", "Special Summon");
        cards.add(new MonsterCard(
            "Dark Magician of Chaos",
            "Wizard with revival powers",
            2,
            owner,
            2800, 2600, 8,
            "DARK", "Spellcaster", "Effect",
            reviveEffect, "Continuous"
        ));

        // Exemple de carte Magie
        Effect spellEffect = new Effect("Revives a monster from the Graveyard", "Special Summon");
        cards.add(new SpellCard(
            "Monster Reborn",
            "Revive a monster",
            3,
            owner,
            "Normal",
            spellEffect,
            false
        ));

        // Exemple de carte Piège
        Effect trapEffect = new Effect("Negates an attack", "Counter");
        cards.add(new TrapCard(
            "Mirror Force",
            "Destroy all attack position monsters your opponent controls",
            4,
            owner,
            "Normal",
            trapEffect,
            false
        ));
    }

    // --- Méthodes principales ---
    public void shuffle() {
        Collections.shuffle(cards);
        System.out.println("Le Deck est mélangé.");
    }

    public Card draw() {
        if (isEmpty()) {
            System.out.println("Le Deck est vide, impossible de piocher.");
            return null;
        }
        Card top = cards.remove(0);
        System.out.println("Carte piochée: " + top.getName());
        return top;
    }

    public void addCard(Card c) {
        cards.add(c);
    }

    public boolean isEmpty() {
        return cards.isEmpty();
    }

    public int size() {
        return cards.size();
    }

    public Card peek() {
        if (isEmpty()) return null;
        return cards.get(0);
    }

    public boolean removeCard(Card c) {
        return cards.remove(c);
    }

    public void reset(List<Card> newCards) {
        this.cards = new ArrayList<>(newCards);
        System.out.println("Le Deck a été réinitialisé.");
    }

    public List<Card> getCards() {
        return cards;
    }
}
