package Deck;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import Card.Card;
import Card.MonsterCard;

/**
 * Classe représentant l’Extra Deck d’un joueur.
 * Contient les cartes Fusion, Synchro, XYZ et Lien.
 */
public class ExtraDeck {

    // --- Attributs ---
    private List<Card> cards;

    // --- Constructeur ---
    public ExtraDeck() {
        this.cards = new ArrayList<>();
    }

    public ExtraDeck(List<Card> initialCards) {
        this.cards = new ArrayList<>(initialCards);
    }

    // --- Méthodes principales ---

    /** Ajoute une carte à l’Extra Deck */
    public void addCard(Card c) {
        cards.add(c);
        System.out.println("Carte ajoutée à l’Extra Deck: " + c.getName());
    }

    /** Retire une carte spécifique de l’Extra Deck */
    public boolean removeCard(Card c) {
        return cards.remove(c);
    }

    /** Sélectionne une carte par son index (utile pour invocation spéciale) */
    public Card chooseCard(int index) {
        if (index < 0 || index >= cards.size()) {
            System.out.println("Index invalide dans l’Extra Deck.");
            return null;
        }
        return cards.get(index);
    }

    /** Mélange les cartes de l’Extra Deck */
    public void shuffle() {
        Collections.shuffle(cards);
        System.out.println("L’Extra Deck est mélangé.");
    }

    /** Retourne le nombre de cartes dans l’Extra Deck */
    public int size() {
        return cards.size();
    }

    /** Vérifie si l’Extra Deck est vide */
    public boolean isEmpty() {
        return cards.isEmpty();
    }

    /** Retourne toutes les cartes de l’Extra Deck */
    public List<Card> getAllCards() {
        return cards;
    }

    /** Recherche les cartes Extra par sous-type (Fusion, Synchro, XYZ, Lien) */
    public List<Card> searchBySubType(String subType) {
        List<Card> result = new ArrayList<>();
        for (Card c : cards) {
            if (c instanceof MonsterCard) {
                MonsterCard m = (MonsterCard) c;
                String st = m.getMonsterType(); // <-- correction
                if (st != null && st.equalsIgnoreCase(subType)) {
                    result.add(c);
                }
            }
        }
        return result;
    }

}
